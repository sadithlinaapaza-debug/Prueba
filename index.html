class Factorial:
    def __init__(self,numero):
        self.numero = numero
        self.resultado = 1

    def calcular(self):
        if self.numero < 0:
            print("El factorial no esta definido para numeros negativos")
            return None
        for i in range(1,self.numero+1):
            self.resultado *= i
        return self.resultado

def main():
        miFactorial = Factorial(5)
        resultado = miFactorial.calcular()
        if resultado is not None:
            print(f"El factorial de {miFactorial.numero} es {resultado}")
if __name__=="__main__":
    main()

class Fibonaci:
    def __init__(self,cantidad):
        self.cantidad = cantidad
        self.serie = []

    def generarSerie(self):
        a,b =0,1
        for _ in range(self.cantidad):
            self.serie.append(a)
            a,b=b,a+b
        return self.serie

def main():
    miFibonaci = Fibonaci(10)
    resultado = miFibonaci.generarSerie()
    print(resultado)
if __name__=="__main__":
    main()

Método de Gradiente

El método del gradiente busca mínimos/ máximos mediante la dirección del gradiente. Ejemplo en R y recursos de descarga abajo.


# MÉTODO DEL GRADIENTE (Ejemplo en R)
f <- function(x){ x^2 } 
f_deriv <- function(x){ 2*x } 
x0 <- 3 
n <- 0.01        # tasa de aprendizaje 
max_iter <- 100 
x_valores <- numeric(max_iter) 
f_valores <- numeric(max_iter) 
for (i in 1:max_iter) { 
  x_nuevo <- x0 - (n * f_deriv(x0))  
  x0 <- x_nuevo 
  x_valores[i] <- x0 
  f_valores[i] <- f(x0) 
} 
cat("El mínimo está en x =", x0, "con f(x) =", f(x0), "\n")

Comparación OLS vs GD

Breve comparación y ejemplo en R.


# Ejemplo abreviado de OLS y GD (R)
set.seed(123)
n <- 1e5
x <- runif(n,0,10)
beta_0 <- 5; beta_1 <- 3
epsilon <- rnorm(n,0,2)
y <- beta_0 + beta_1 * x + epsilon
# OLS
modelo_ols <- lm(y ~ x)
summary(modelo_ols)
# GD (simplificado)
alpha <- 1e-6; iter <- 1000
b0 <- 0; b1 <- 0
for(i in 1:iter){
  y_pred <- b0 + b1*x
  grad_b0 <- (-2/length(x)) * sum(y - y_pred)
  grad_b1 <- (-2/length(x)) * sum(x * (y - y_pred))
  b0 <- b0 - alpha * grad_b0
  b1 <- b1 - alpha * grad_b1
}
print(c(b0,b1))
